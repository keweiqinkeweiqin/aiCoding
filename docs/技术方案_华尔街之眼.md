```java
/**
 * 数据采集调度器，聚合多个数据源采集器
 * 采集流程：拉取原始数据 → 标准化为 IntelItem → 质量把控 → 入库
 */
@Service
public class DataCollectorService {

    @Autowired private NewsApiCollector newsCollector;
    @Autowired private AlphaVantageCollector alphaCollector;
    @Autowired private RedditCollector redditCollector;
    @Autowired private QualityControlService qualityControl;
    @Autowired private IntelItemRepository repository;

    /**
     * 执行全量采集
     * 1. 并行调用各数据源
     * 2. 标准化为 IntelItem
     * 3. 质量把控（去重、置信度、可信度）
     * 4. 交叉验证
     * 5. 批量入库
     */
    public int collectAll() {
        List<IntelItem> allItems = new ArrayList<>();
        allItems.addAll(newsCollector.collect());      // 新闻
        allItems.addAll(alphaCollector.collect());      // 行情+新闻
        allItems.addAll(redditCollector.collect());     // 社交

        // 质量把控
        List<IntelItem> processed = qualityControl.process(allItems);

        // 批量入库（去重后）
        return repository.saveAll(processed).size();
    }
}
```

```java
     * 来源可信度标注规则
     */
    private void assignCredibility(IntelItem item) {
        // 权威来源列表
        Set<String> authoritative = Set.of("Reuters", "Bloomberg", "CNBC",
            "SEC Filing", "官方财报");
        // 存疑来源列表
        Set<String> doubtful = Set.of("Reddit", "匿名论坛", "未知来源");

        if (authoritative.contains(item.getSourceName())) {
            item.setCredibility(Credibility.AUTHORITATIVE);
        } else if (doubtful.contains(item.getSourceName())) {
            item.setCredibility(Credibility.DOUBTFUL);
        } else {
            item.setCredibility(Credibility.NORMAL);
        }
    }

    /**
     * 置信度评估（LLM辅助 + 规则引擎）
     * 综合评分 = 来源权威性(40%) + 信息一致性(30%) + 时效性(20%) + 完整性(10%)
     */
    private void assessConfidence(IntelItem item) {
        double score = 0.0;

        // 来源权威性 (40%)
        switch (item.getCredibility()) {
            case AUTHORITATIVE: score += 0.4 * 1.0; break;
            case NORMAL:        score += 0.4 * 0.6; break;
            case DOUBTFUL:      score += 0.4 * 0.2; break;
        }

        // 时效性 (20%) — 指数衰减
        long hoursAgo = ChronoUnit.HOURS.between(item.getPublishedAt(), LocalDateTime.now());
        score += 0.2 * Math.exp(-0.05 * hoursAgo);

        // 完整性 (10%) — 有正文内容加分
        if (item.getContent() != null && item.getContent().length() > 100) {
            score += 0.1;
        }

        // 信息一致性 (30%) — 交叉验证后补充
```

```java
    /**
     * 同步调用 LLM
     */
    public String chat(String systemPrompt, String userMessage) {
        String url = llmConfig.getBaseUrl() + "/v1/chat/completions";

        Map<String, Object> body = Map.of(
            "model", llmConfig.getModel(),
            "messages", List.of(
                Map.of("role", "system", "content", systemPrompt),
                Map.of("role", "user", "content", userMessage)
            ),
            "temperature", 0.7,
            "max_tokens", 2000
        );

        HttpHeaders headers = new HttpHeaders();
        headers.setBearerAuth(llmConfig.getApiKey());
        headers.setContentType(MediaType.APPLICATION_JSON);

        ResponseEntity<Map> response = restTemplate.exchange(
            url, HttpMethod.POST,
            new HttpEntity<>(body, headers),
            Map.class
        );
```

```java
/**
 * 个性化研判生成
 * 核心逻辑：用户画像 + 情报摘要 → Prompt → LLM → 结构化研判
 */
@Service
public class AnalysisService {

    @Autowired private LlmService llmService;
    @Autowired private UserProfileRepository profileRepo;
    @Autowired private IntelItemRepository intelRepo;
    @Autowired private AnalysisRecordRepository analysisRepo;

    private static final String SYSTEM_PROMPT =
        "你是一位专业的AI/科技领域投资分析师。请基于用户画像和最新情报，" +
        "生成结构化的投资研判。输出格式为JSON：\n" +
        "{\n" +
        "  \"analysis\": \"情报研判（200字内）\",\n" +
        "  \"impacts\": [{\"stock\": \"代码\", \"impact\": \"影响分析\"}],\n" +
        "  \"suggestion\": \"操作建议\",\n" +
        "  \"risks\": [\"风险点1\", \"风险点2\"]\n" +
        "}";

    public AnalysisResponse generate(Long userId, List<Long> intelIds) {
        UserProfile profile = profileRepo.findById(userId).orElseThrow();
        List<IntelItem> intels = intelRepo.findAllById(intelIds);

        String userMessage = buildUserMessage(profile, intels);
        String result = llmService.chat(SYSTEM_PROMPT, userMessage);

        // 解析 JSON 结果
        AnalysisResponse response = parseAnalysis(result);

        // 存储研判记录
        saveRecord(userId, intelIds, result, profile.getInvestmentStyle());

        return response;
    }

    /**
     * 流式生成（SSE）
     */
    public void generateStream(Long userId, List<Long> intelIds,
                                SseEmitter emitter) {
        UserProfile profile = profileRepo.findById(userId).orElseThrow();
        List<IntelItem> intels = intelRepo.findAllById(intelIds);

        String userMessage = buildUserMessage(profile, intels);
        llmService.chatStream(SYSTEM_PROMPT, userMessage, emitter);
    }

    private String buildUserMessage(UserProfile profile, List<IntelItem> intels) {
        StringBuilder sb = new StringBuilder();
        sb.append("用户画像：\n");
        sb.append("- 关注领域：").append(profile.getFocusAreas()).append("\n");
        sb.append("- 投资风格：").append(profile.getInvestmentStyle()).append("\n");
        sb.append("- 风险偏好：").append(profile.getRiskTolerance()).append("\n");
        sb.append("- 投资周期：").append(profile.getInvestmentHorizon()).append("\n");
        sb.append("- 当前持仓：\n");
        for (Holding h : profile.getHoldings()) {
            sb.append("  ").append(h.getStockCode())
              .append(" ").append(h.getQuantity()).append("股")
              .append(" 成本$").append(h.getAvgCost()).append("\n");
        }
        sb.append("\n最新情报：\n");
        for (IntelItem intel : intels) {
            sb.append("- [").append(intel.getCredibility()).append("] ")
              .append(intel.getTitle())
              .append(" (来源:").append(intel.getSourceName())
              .append(", 置信度:").append(intel.getConfidenceScore()).append(")\n");
            if (intel.getSummary() != null) {
                sb.append("  摘要：").append(intel.getSummary()).append("\n");
            }
        }
        return sb.toString();
```

```javascript
// api/analysis.js
export function generateAnalysisStream(userId, intelIds, onChunk, onDone) {
  const url = `/api/analysis/stream?userId=${userId}&intelIds=${intelIds.join(',')}`;
  const eventSource = new EventSource(url);

  eventSource.onmessage = (event) => {
    if (event.data === '[DONE]') {
      eventSource.close();
      onDone();
      return;
    }
    onChunk(event.data);  // 逐段追加到页面
  };

  eventSource.onerror = () => {
    eventSource.close();
  };

  return eventSource;
}
```

## 六、数据采集器实现方案

### 6.1 NewsAPI 采集器

```java
@Component
public class NewsApiCollector {

    @Value("${datasource.newsapi.key}")
    private String apiKey;

    @Autowired private RestTemplate restTemplate;

    private static final String BASE_URL = "https://newsapi.org/v2/everything";

    public List<IntelItem> collect() {
        String url = BASE_URL + "?q=AI+OR+NVIDIA+OR+semiconductor+OR+OpenAI"
            + "&language=en&sortBy=publishedAt&pageSize=50"
            + "&apiKey=" + apiKey;

        NewsApiResponse resp = restTemplate.getForObject(url, NewsApiResponse.class);

        return resp.getArticles().stream()
            .map(article -> IntelItem.builder()
                .title(article.getTitle())
                .summary(article.getDescription())
                .content(article.getContent())
                .sourceType(SourceType.NEWS)
                .sourceName(article.getSource().getName())
                .sourceUrl(article.getUrl())
                .publishedAt(parseDate(article.getPublishedAt()))
                .collectedAt(LocalDateTime.now())
                .contentHash(md5(article.getTitle() + article.getDescription()))
                .build())
            .collect(Collectors.toList());
    }
}
```

## 九、性能与稳定性

| 场景 | 方案 |
|------|------|
| LLM 响应慢（>10s） | 设置 30s 超时 + 前端 loading 状态 + 热门情报预生成研判缓存 |
| 外部 API 限流 | 本地缓存最近一次采集结果，限流时返回缓存 |
| 并发请求 | Spring Boot 默认 Tomcat 200 线程，比赛场景足够 |
| 数据量增长 | H2 文件模式，定期清理 30 天前的旧数据 |
| 异常处理 | 全局 @ControllerAdvice 统一异常返回 |
